
command for listing configuration list

    >git config --list

command for adding user name 

    >git config --global user.name "username"

command for adding user email
    >git config --global user.email "user@gmail.com"

command for initializing git 
    >git init


command for listing all untracked files 

    >git status

command for adding modified files to staging area

    >git add -A

command for adding all staged files to localrepository

    >git commit -m "commit message"

flags
    A => staged
    U => untracked
    M => modified
